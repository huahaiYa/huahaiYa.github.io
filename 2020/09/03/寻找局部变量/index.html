寻找局部变量

对c语言函数逆向

寻找函数中的局部变量以及参数的位置

小程序

int fun(int a, int b, int c)
{

	int x = 2;
	int y = 3;
	int z = 4;

	return a + b + c + x + y + z;

}


int main(void)
{
	
	
	fun(7, 8, 9);



	system("pause");
	return 0;



}

在fun函数处下断点，在vs2019中进行反调试

push        9  
push        8  
push        7  
call        __vfprintf_l (06B13B6h)  
add         esp,0Ch

在fun函数入口处发现push操作 此时的esp寄存器地址为ESP = 004FFA34，f11单步进入call内，此时esp寄存器地址为ESP = 004FFA24，查看call内代码

push ebp
mov ebp,esp
sub esp,0E4h

此处提升了堆栈空间，将ebp的值和esp相等，所以参数的内存位置在esp-4.esp-8以及esp-c的位置

push ebx
push esi 保存堆栈空间
push edi
lea edi,[ebp-0E4h]
mov ecx,39h
mov eax,0CCCCCCCCh 填充数据
rep stos dword ptr es:[edi]
mov ecx,offset _AC7D5F78_多文件@c (06BC010h)
call @__CheckForDebuggerJustMyCode@4 (06B1212h)

mov dword ptr [x],2

mov dword ptr [y],3 这里保存了局部变量的值，直接放到了内存中，后面直接进行了计算

mov dword ptr [z],4 

return a + b + c + x + y + z;

mov eax,dword ptr [a]
add eax,dword ptr [b]
add eax,dword ptr [c]
add eax,dword ptr [x]
add eax,dword ptr [y]
add eax,dword ptr [z] 返回值放到了eax中

}
pop edi
pop esi
pop ebx
add esp,0E4h
cmp ebp,esp
call __RTC_CheckEsp (06B121Ch)
mov esp,ebp
pop ebp
ret

总结

在函数中对反汇编代码进行查看发现局部变量的值直接存放在内存中进行存储，后面直接对内存进行访问取值，但是参数的值是通过入栈的方式，并且其参数一般存放在了寄存器ebp+4的地址中
